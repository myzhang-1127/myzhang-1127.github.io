58 result_type result(Args const &args) const
59 {
60 return numeric::fdiv(
61 accumulators::moment<3>(args)

63 + 2. * mean(args) * mean(args) * mean(args)
64 , ( accumulators::moment<2>(args) - mean(args) * mean(args) )
