107 // droppable_accumulator_base
108 template<typename Accumulator>
109 struct droppable_accumulator_base
110 : Accumulator
111 {
112 typedef droppable_accumulator_base base;
113 typedef mpl::true_ is_droppable;
114 typedef typename Accumulator::result_type result_type;
115
116 template<typename Args>
117 droppable_accumulator_base(Args const &args)
118 : Accumulator(args)
119 , ref_count_(0)
120 {
121 }
122
123 droppable_accumulator_base(droppable_accumulator_base const &that)
124 : Accumulator(*static_cast<Accumulator const *>(&that))
125 , ref_count_(that.ref_count_)
126 {
127 }
128
129 template<typename Args>
130 void operator ()(Args const &args)
131 {
132 if(!this->is_dropped())
133 {
134 this->Accumulator::operator ()(args);
135 }
136 }
137
138 template<typename Args>
139 void add_ref(Args const &)
140 {
141 ++this->ref_count_;
142 }
143
144 template<typename Args>
145 void drop(Args const &args)
146 {
147 BOOST_ASSERT(0 < this->ref_count_);
148 if(1 == this->ref_count_)
149 {
150 static_cast<droppable_accumulator<Accumulator> *>(this)->on_drop(args);
151 }
152 --this->ref_count_;
153 }
154
155 bool is_dropped() const
156 {
157 return 0 == this->ref_count_;
158 }
159
160 private:
161 int ref_count_;
162 };

184 // with_cached_result
185 template<typename Accumulator>
186 struct with_cached_result
187 : Accumulator
188 {
189 typedef typename Accumulator::result_type result_type;
190
191 template<typename Args>
192 with_cached_result(Args const &args)
193 : Accumulator(args)
194 , cache()
195 {
196 }
197
198 with_cached_result(with_cached_result const &that)
199 : Accumulator(*static_cast<Accumulator const *>(&that))
200 , cache()
201 {
202 if(that.has_result())
203 {
204 this->set(that.get());
205 }
206 }
207
208 ~with_cached_result()
209 {
210 // Since this is a base class of droppable_accumulator_base,
211 // this destructor is called before any of droppable_accumulator_base's
212 // members get cleaned up, including is_dropped, so the following
213 // call to has_result() is valid.
214 if(this->has_result())
215 {
216 this->get().~result_type();
217 }
218 }
219
220 template<typename Args>
221 void on_drop(Args const &args)
222 {
223 // cache the result at the point this calculation was dropped
224 BOOST_ASSERT(!this->has_result());
225 this->set(this->Accumulator::result(args));
226 }
227
228 template<typename Args>
229 result_type result(Args const &args) const
230 {
231 return this->has_result() ? this->get() : this->Accumulator::result(args);
232 }
233
234 private:
235 BOOST_DELETED_FUNCTION(with_cached_result &operator =(with_cached_result const &))
236
237 void set(result_type const &r)
238 {
239 ::new(this->cache.address()) result_type(r);
240 }
241
242 result_type const &get() const
243 {
244 return *static_cast<result_type const *>(this->cache.address());
245 }
246
247 bool has_result() const
248 {
249 typedef with_cached_result<Accumulator> this_type;
250 typedef droppable_accumulator_base<this_type> derived_type;
251 return static_cast<derived_type const *>(this)->is_dropped();
252 }
253
254 aligned_storage<sizeof(result_type)> cache;
255 };
